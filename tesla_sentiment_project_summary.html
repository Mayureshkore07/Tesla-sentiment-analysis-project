# Handle missing dates for datetime creation
def convert_date(x):
    x = str(x).strip()
    if 'Today' in x:
        return datetime.datetime.today().strftime('%Y-%m-%d')
    elif 'Yesterday' in x:
        return (datetime.datetime.today() - datetime.timedelta(days=1)).strftime('%Y-%m-%d')
    else:
        return x

sentiment_df['date_clean'] = sentiment_df['date'].apply(convert_date)
sentiment_df['datetime'] = pd.to_datetime(sentiment_df['date_clean'] + ' ' + sentiment_df['time'], errors='coerce')
sentiment_df = sentiment_df.dropna(subset=['datetime'])

# Daily Sentiment Summary
daily_sentiment = sentiment_df.groupby([sentiment_df['datetime'].dt.date, 'sentiment_label']).size().unstack().fillna(0)
daily_sentiment.to_csv("outputs/daily_sentiment_summary.csv")

# Plot Daily Sentiment
daily_sentiment.plot(kind='bar', stacked=True, figsize=(12,6), color=['green','red','blue'])
plt.title("Daily Sentiment Summary")
plt.xlabel("Date")
plt.ylabel("Number of News")
plt.xticks(rotation=45)
plt.tight_layout()
plt.savefig("outputs/phase4_daily_sentiment_summary.png")
plt.close()
